## The problem

Each review item must be resolved individually — click a verdict button, wait for the response, page reloads (or item disappears if prompt 06 has been applied), find the next item. When there are dozens of items in the queue, many with the same aggregator verdict (e.g., all leaning "cohere"), confirming each one individually is tedious. There is no way to select multiple items and confirm or resolve them in batch.

## Your task

Add batch selection and batch resolution to the review queue. Users should be able to select multiple items, then apply a single verdict to all selected items.

5. **Create the batch resolution function**:

   ```typescript
   async function batchResolve(verdict: "cohere" | "contradict" | "unresolved", overrule: boolean) {
     setMsg("");
     const ids = Array.from(selected);
     const res = await fetch("/api/review/batch", {
       method: "POST",
       headers: { "Content-Type": "application/json" },
       body: JSON.stringify({ ids, verdict, overrule, note: `Batch ${overrule ? "overrule" : "confirm"} via portal` }),
     });
     if (!res.ok) {
       const data = await res.json();
       setMsg(data.error || "Batch resolution failed");
       return;
     }
     // Remove resolved items from view (if prompt 06 pattern is applied)
     setResolved((prev) => { const next = new Set(prev); ids.forEach((id) => next.add(id)); return next; });
     setSelected(new Set());
     setLastAction({ id: "batch", verdict });
   }
   ```

6. **Create `api/review/batch/route.ts`**:
   ```typescript
   import { NextResponse } from "next/server";
   import { getFounder } from "@/lib/auth";
   import { db } from "@/lib/db";
   import { pushReviewResolutionToNoosphere } from "@/lib/pushReviewToNoosphere";

   export async function POST(req: Request) {
     const founder = await getFounder();
     if (!founder) return NextResponse.json({ error: "Unauthorized" }, { status: 401 });

     const { ids, verdict, overrule, note } = await req.json() as {
       ids: string[];
       verdict: "cohere" | "contradict" | "unresolved";
       overrule: boolean;
       note: string;
     };

     if (!ids?.length || !verdict) {
       return NextResponse.json({ error: "ids and verdict required" }, { status: 400 });
     }

     // Update all items in a transaction
     await db.$transaction(
       ids.map((id) =>
         db.reviewItem.update({
           where: { id },
           data: {
             status: "done",
             humanVerdict: verdict,
             humanOverrule: overrule,
             resolutionNote: note,
             resolvedAt: new Date(),
             resolvedByFounderId: founder.id,
           },
         })
       )
     );

     // Sync to Noosphere (fire-and-forget for batch)
     const items = await db.reviewItem.findMany({ where: { id: { in: ids } } });
     for (const item of items) {
       if (item.noosphereId) {
         pushReviewResolutionToNoosphere({
           reviewId: item.noosphereId,
           verdict,
           overrule,
           aggregatorVerdict: item.aggregatorVerdict,
           founderId: founder.noosphereId || founder.id,
           note,
         }).catch(() => {}); // fire-and-forget
       }
     }

     return NextResponse.json({ ok: true, count: ids.length });
   }
   ```
