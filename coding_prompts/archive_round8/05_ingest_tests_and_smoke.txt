def test_m4a_upload_no_longer_raises_textcontent_error(
    fake_codex_db, upload_factory, monkeypatch, tiny_audio_fixture

### 4. `test_ingest_status_transitions.py`

Tracks status transitions by listening to DB UPDATEs. Uses a sqlite trigger:

```sql
CREATE TABLE "UploadStatusHistory" (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  upload_id TEXT, status TEXT, ts DATETIME DEFAULT CURRENT_TIMESTAMP
);
CREATE TRIGGER upload_status_log
  AFTER UPDATE OF status ON "Upload"
  FOR EACH ROW
  BEGIN
    INSERT INTO "UploadStatusHistory" (upload_id, status) VALUES (NEW.id, NEW.status);
  END;
```

After running the pipeline on a fixture, assert the history contains the ordered subsequence:

```python
expected = ["extracting", "awaiting_ingest", "processing", "ingested"]
```

For failed paths, assert the history ends with `failed` and the relevant intermediate states are present up to the failure point.

### 5. `e2e/test_ingest_audio_smoke.py` — real transcription

```python
import os
import pytest

@pytest.mark.skipif(
    os.environ.get("NOOSPHERE_E2E_SMOKE") != "1",
    reason="gated smoke; set NOOSPHERE_E2E_SMOKE=1 to enable",
)
def test_real_m4a_end_to_end(fake_codex_db, upload_factory):
    """Actually transcribes a short fixture .m4a with faster-whisper,
    runs the full pipeline, and asserts Conclusions were written."""
    fixture = Path(__file__).parent.parent / "fixtures" / "smoke_meeting_30s.m4a"
    if not fixture.exists():
        pytest.skip(f"{fixture} not present (generate with scripts/make_smoke_audio.py)")
    uid = upload_factory(mime="audio/mp4", text=None, file_path=str(fixture))
    from noosphere.codex_bridge import ingest_from_codex
    result = ingest_from_codex(upload_id=uid, use_llm=False, dry_run=False,
                               codex_db_url=sqlite_url_for(fake_codex_db))
    assert result.num_conclusions_written > 0
    # spot check a known phrase from the smoke clip
    rows = fake_codex_db.execute(
        'SELECT "conclusionText" FROM "Conclusion" WHERE "organizationId"=?',
        ("org_1",),
    ).fetchall()
    assert any("purpose" in r[0].lower() for r in rows), \
        "expected the word 'purpose' somewhere in transcribed conclusions"
```

Include `scripts/make_smoke_audio.py` that synthesizes a 30s clip via macOS `say` + `ffmpeg`. Not run in CI; developers use it once.
