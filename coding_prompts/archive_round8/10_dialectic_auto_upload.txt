## Step-by-step

### 1. Codex API changes

In `prepare/route.ts`, accept these new body fields (all optional):

```ts
transcript?: string;
extractionMethod?: string;
recordedDate?: string;  // ISO date
```

On insert, populate the Upload row:

```ts
textContent: body.transcript ?? null,
extractionMethod: body.extractionMethod ?? null,
sourceType: body.transcript ? "transcript" : "audio",
status: body.transcript ? "awaiting_ingest" : "pending",
```

When `textContent` is populated at prepare-time, status jumps straight to `awaiting_ingest` so the Codex dashboard reflects that extraction is already done. `ingest-from-codex` picks it up from there and runs only the claim-extraction stage.

The `finalize` route can also accept `transcript` to allow late-arriving override (rare — only if Dialectic re-transcribed post-upload).

### 2. `codex_upload.py`

```python
from __future__ import annotations
import json, time
from dataclasses import dataclass
from pathlib import Path
from typing import Callable, Optional

import requests

from dialectic.config import AutoUploadConfig


@dataclass(frozen=True)
class UploadResult:
    upload_id: str
    codex_url: str
    bytes_sent: int


class UploadError(Exception): pass


def upload_recording(
    *,
    audio_path: Path,
    transcript: str,
    title: str,
    recorded_date: str,
    codex_url: str,
    api_key: str,
    extraction_method: str = "dialectic-faster-whisper",
    on_progress: Optional[Callable[[int, int], None]] = None,
    cfg: AutoUploadConfig | None = None,
) -> UploadResult:
    cfg = cfg or AutoUploadConfig()
    size = audio_path.stat().st_size

    # 1. prepare
    prep = requests.post(
        f"{codex_url.rstrip('/')}/api/upload/audio/prepare",
        headers={"Authorization": f"Bearer {api_key}", "Content-Type": "application/json"},
        json={
            "filename": audio_path.name,
            "mimeType": "audio/wav",
            "fileSize": size,
            "title": title,
            "recordedDate": recorded_date,
            "transcript": transcript,
            "extractionMethod": extraction_method,
        },
        timeout=cfg.prepare_timeout_seconds,
    )
    if not prep.ok:
        raise UploadError(f"prepare failed: {prep.status_code} {prep.text}")
    prep_body = prep.json()
    upload_id = prep_body["uploadId"]
    signed_put = prep_body["signedPutUrl"]

    # 2. PUT bytes
    with audio_path.open("rb") as f:
        # Use a chunked upload so we can report progress.
        sent = 0
        def iter_chunks():
            nonlocal sent
            while True:
                chunk = f.read(cfg.chunk_bytes)
                if not chunk:
                    break
                sent += len(chunk)
                if on_progress:
                    on_progress(sent, size)
                yield chunk
        put = requests.put(
            signed_put, data=iter_chunks(),
            headers={"Content-Type": "audio/wav", "Content-Length": str(size)},
            timeout=cfg.put_timeout_seconds,
        )
    if put.status_code not in (200, 201, 204):
        raise UploadError(f"PUT to signed URL failed: {put.status_code} {put.text[:200]}")

    # 3. finalize
    fin = requests.post(
        f"{codex_url.rstrip('/')}/api/upload/audio/finalize/{upload_id}",
        headers={"Authorization": f"Bearer {api_key}", "Content-Type": "application/json"},
        json={"fileSize": size, "transcript": transcript,
              "extractionMethod": extraction_method},
        timeout=cfg.finalize_timeout_seconds,
    )
    if not fin.ok:
        raise UploadError(f"finalize failed: {fin.status_code} {fin.text}")

    return UploadResult(
        upload_id=upload_id,
        codex_url=f"{codex_url.rstrip('/')}/dashboard/uploads/{upload_id}",
        bytes_sent=size,
    )
```

### 3. `pending_queue.py`

```python
from __future__ import annotations
import json, os, platform, tempfile
from dataclasses import dataclass, asdict
from pathlib import Path


def queue_dir() -> Path:
    if platform.system() == "Darwin":
        root = Path.home() / "Library" / "Application Support" / "Theseus" / "Dialectic"
    elif platform.system() == "Windows":
        root = Path(os.environ.get("APPDATA", Path.home())) / "Theseus" / "Dialectic"
    else:
        root = Path(os.environ.get("XDG_DATA_HOME", Path.home() / ".local" / "share")) / "theseus" / "dialectic"
    root.mkdir(parents=True, exist_ok=True)
    return root


def queue_file() -> Path:
    return queue_dir() / "pending_uploads.jsonl"


@dataclass
class PendingUpload:
    audio_path: str
    transcript: str
    title: str
    recorded_date: str
    extraction_method: str
    created_at: str

def enqueue(p: PendingUpload) -> None:
    # Atomic append: write the record to a temp file beside the queue file, then
    # append-move to avoid a partial line if the process is killed mid-write.
    line = json.dumps(asdict(p), ensure_ascii=False) + "\n"
    with queue_file().open("a", encoding="utf-8") as f:
        f.write(line)


def drain(*, upload_fn) -> dict[str, int]:
    """Called by a menu action or a retry helper. Processes each line; on success
    removes it. On failure, leaves the rest intact for the next run."""
    path = queue_file()
    if not path.exists():
        return {"processed": 0, "failed": 0}
    lines = path.read_text(encoding="utf-8").splitlines()
    remaining: list[str] = []
    ok = fail = 0
    for line in lines:
        if not line.strip():
            continue
        try:
            p = PendingUpload(**json.loads(line))
            upload_fn(p)
            ok += 1
        except Exception as e:
            remaining.append(line)
            fail += 1
    path.write_text("\n".join(remaining) + ("\n" if remaining else ""), encoding="utf-8")
    return {"processed": ok, "failed": fail}
```

### 4. Pipeline replacement

```python
def _stage_upload(self):
    self.stage_started.emit("upload")
    from dialectic.codex_upload import upload_recording, UploadError
    from dialectic.credentials import load_cloud_credentials
    from dialectic.pending_queue import enqueue, PendingUpload
    from datetime import datetime, timezone

    creds = load_cloud_credentials()
    if not creds:
        self.stage_failed.emit("upload", "no Codex credentials configured")
        return

    audio_path = self.result.trimmed_audio or self.artifact.audio_path
    try:
        result = upload_recording(
            audio_path=audio_path,
            transcript=self.result.transcript or "",
            title=self.result.title or "",
            recorded_date=self.result.recorded_date or "",
            codex_url=creds.url,
            api_key=creds.api_key,
            extraction_method=f"dialectic-{self.result.transcript_model or 'faster-whisper'}",
            on_progress=lambda sent, total: self.stage_progress.emit(
                "upload", int(sent * 100 / max(total, 1))
            ),
        )
        self.result.upload_id = result.upload_id
        self.result.codex_url = result.codex_url
        self.stage_succeeded.emit("upload", result)
    except UploadError as e:
        # Queue for retry and tell the user.
        enqueue(PendingUpload(
            audio_path=str(audio_path),
            transcript=self.result.transcript or "",
            title=self.result.title or "",
            recorded_date=self.result.recorded_date or "",
            extraction_method=f"dialectic-{self.result.transcript_model or 'faster-whisper'}",
            created_at=datetime.now(timezone.utc).isoformat(),
        ))
        self.stage_failed.emit("upload",
            f"{e} — queued for retry (Dialectic menu → 'Retry pending uploads')")
```

## Prohibitions

- Do NOT upload the untrimmed audio. The trimmed .wav is what Dialectic promises to upload.
- Do NOT store the transcript in a separate Upload field ("transcript" column). Reuse `textContent`; that's the column `ingest-from-codex` reads. Divergence between `transcript` and `textContent` silently breaks ingest.
- Do NOT require internet connectivity during recording. Everything through trim/transcribe/title must work offline; only this stage needs the network, and it gracefully queues on failure.
- Do NOT store the `DIALECTIC_CLOUD_API_KEY` in a new location. Reuse `credentials.py`.
- Do NOT skip the `pending_uploads.jsonl` queue even for "transient" errors — a flaky wifi moment is the common case.
