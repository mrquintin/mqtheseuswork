### 1. `BatchTranscriptionConfig`

```python
@dataclass
class BatchTranscriptionConfig:
    model: str = "medium.en"
    compute_type: str = "int8"         # override via env DIALECTIC_WHISPER_COMPUTE_TYPE
    beam_size: int = 5                 # larger beam than live; batch has no latency pressure
    language: str | None = "en"
    vad_filter: bool = True            # drop silence segments from Whisper's own output
    initial_prompt: str | None = (
        "This is a Theseus dialectic recording — a spoken discussion among "
        "founders about methodology, truth-tracking, and epistemic practice."
    )
```

The `initial_prompt` biases Whisper toward Theseus-relevant vocabulary (methodology, epistemic, discount, coherence…). Measure before and after on a founder recording; the bias typically raises accuracy ~1-3 WER points on unusual Theseus terms. Document that this is worth auditing if you ever switch domains.

### 2. `batch_transcriber.py`

```python
from __future__ import annotations
from dataclasses import dataclass, field
from pathlib import Path
import os, time
from typing import Optional

from dialectic.config import BatchTranscriptionConfig


@dataclass(frozen=True)
class TranscriptSegment:
    start_s: float
    end_s: float
    text: str


@dataclass(frozen=True)
class TranscriptionResult:
    text: str
    segments: list[TranscriptSegment]
    language: str
    model_name: str
    duration_seconds: float   # audio duration
    elapsed_seconds: float    # transcription wall clock


_MODEL_CACHE: dict[str, object] = {}


def _get_model(name: str, compute_type: str):
    key = f"{name}:{compute_type}"
    if key not in _MODEL_CACHE:
        from faster_whisper import WhisperModel
        _MODEL_CACHE[key] = WhisperModel(name, device="cpu", compute_type=compute_type)
    return _MODEL_CACHE[key]


def transcribe(path: Path, cfg: BatchTranscriptionConfig | None = None) -> TranscriptionResult:
    cfg = cfg or BatchTranscriptionConfig()
    compute_type = os.environ.get("DIALECTIC_WHISPER_COMPUTE_TYPE", cfg.compute_type)
    model = _get_model(cfg.model, compute_type)
    t0 = time.monotonic()
    segments_iter, info = model.transcribe(
        str(path),
        language=cfg.language,
        beam_size=cfg.beam_size,
        vad_filter=cfg.vad_filter,
        initial_prompt=cfg.initial_prompt,
    )
    segments = [
        TranscriptSegment(start_s=s.start, end_s=s.end, text=s.text.strip())
        for s in segments_iter if s.text and s.text.strip()
    ]
    text = " ".join(s.text for s in segments).strip()
    return TranscriptionResult(
        text=text, segments=segments,
        language=info.language or cfg.language or "unknown",
        model_name=cfg.model,
        duration_seconds=info.duration or 0.0,
        elapsed_seconds=time.monotonic() - t0,
    )
```

### 3. Pipeline replacement

In `recording_pipeline.py`, replace `_stage_transcribe`:

```python
def _stage_transcribe(self):
    self.stage_started.emit("transcribe")
    try:
        from dialectic.batch_transcriber import transcribe
        source = self.result.trimmed_audio or self.artifact.audio_path
        result = transcribe(source)
        self.result.transcript = result.text
        self.result.transcript_segments = result.segments
        self.result.transcript_language = result.language
        self.result.transcript_model = result.model_name
        self.stage_succeeded.emit("transcribe", result)
    except Exception as e:
        self.stage_failed.emit("transcribe", f"{type(e).__name__}: {e}")
        raise
```

Add the new fields to `PipelineResult`:

```python
transcript_segments: list = field(default_factory=list)
transcript_language: Optional[str] = None
transcript_model: Optional[str] = None
```
