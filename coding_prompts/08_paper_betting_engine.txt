```python
class PaperBetConfig:
    edge_threshold: float = 0.05          # |model_p - market_p| < this ‚Üí no bet
    kelly_fraction: float = 0.25          # fractional Kelly
    max_stake_usd: float = 50.0           # ceiling per bet (PAPER)
    initial_balance_usd: float = 10_000.0
    organization_id: str = ""

    @classmethod
    def from_env(cls) -> "PaperBetConfig": ...
```

Env keys: `FORECASTS_PAPER_EDGE_THRESHOLD`, `FORECASTS_PAPER_KELLY_FRACTION`, `FORECASTS_PAPER_MAX_STAKE_USD`, `FORECASTS_PAPER_INITIAL_BALANCE_USD`, `FORECASTS_INGEST_ORG_ID`.

```python
async def evaluate_and_stake(
    store, prediction_id: str, *, config: PaperBetConfig, now: datetime | None = None,
) -> Optional[ForecastBet]:
    """
    Look up the prediction + its market.
      1. Compute edge = (probability_yes - market_yes_price). If
         abs(edge) < config.edge_threshold ‚Üí no bet (return None).
      2. Side := YES if edge > 0 else NO.
      3. Kelly fraction-of-bankroll:
            f* = (b*p - q) / b
         where p = model probability of side winning, q = 1-p, and
         b = (1 - market_price_for_side) / market_price_for_side. Clamp to [0, 1].
      4. Stake := min(f* * kelly_fraction * paper_balance, max_stake_usd).
         If stake < $1 ‚Üí no bet.
      5. Persist a ForecastBet row with mode=PAPER, status=PAPER_FILLED, entryPrice
         set to the *market's* current price for the chosen side, stakeUsd, side.
      6. Decrement paper_balance by stakeUsd. Return the row.
    """

async def settle_paper_bets_for_market(store, market_id: str) -> int:
    """
    On market resolution: for every PAPER ForecastBet on this market with
    status=PAPER_FILLED, compute payoff = (stake / entryPrice) if side won else 0,
    set status=SETTLED, exitPrice = 1.0 if won else 0.0, P&L = payoff - stake.
    Increment paper_balance by payoff. Return count of settled bets.
    """
```

### Wiring

- `forecast_generator.generate_forecast(...)` calls `evaluate_and_stake(...)` after a successful PUBLISHED outcome. If `is_live_authorized(...)` is False (prompt 09 introduces this; for now, always False), only paper is fired.
- `resolution_tracker.poll_market(...)` calls `settle_paper_bets_for_market(...)` after writing the ForecastResolution.
