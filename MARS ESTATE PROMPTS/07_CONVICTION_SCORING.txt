================================================================================
07 — Conviction scoring
================================================================================

GOAL
Maintain a 0..1 conviction value on every active Principle. Drives Oracle
retrieval weighting, sort order on /principles, decay, and the
PrinciplesPanel on the home page. Operator can pin a value to override
the formula.

TASK

1. Create lib/engine/extract/conviction.ts:

  import { Claims, Principles, DriftEvents } from '@/lib/db';
  import { recordEvent } from '@/lib/audit/recordEvent';
  import { ACTIONS } from '@/lib/audit/actions';

  const MODALITY_WEIGHT: Record<string, number> = {
    assertion:    1.0,
    prediction:   0.7,
    counter:      0.6,
    hypothetical: 0.3,
    question:     0.1,
  };

  // Operator allowlist + email domain heuristic. Adjust as we learn from
  // Colin's actual speaker labels. Stored in Meta so the operator can
  // edit at runtime via /settings/extractor (prompt 27).
  const OPERATOR_NAMES_DEFAULT = ['colin', 'colin yuan'];
  const OPERATOR_EMAIL_DOMAIN_DEFAULT = '@marsestate';

  function getOperatorIdentity() {
    // Read from Meta with defaults — keeps tunable without redeploy.
    const Meta = require('@/lib/db').Meta as { get: (k: string) => string | null };
    const names = (Meta.get('operator_speaker_names') ?? OPERATOR_NAMES_DEFAULT.join(',')).split(',').map(s => s.trim().toLowerCase());
    const domain = (Meta.get('operator_email_domain') ?? OPERATOR_EMAIL_DOMAIN_DEFAULT).toLowerCase();
    return { names, domain };
  }

  function speakerAuthority(speaker: string | null): number {
    if (!speaker) return 0.5;
    const lower = speaker.toLowerCase();
    const { names, domain } = getOperatorIdentity();
    if (names.some(n => lower.includes(n))) return 1.0;
    if (lower.includes(domain)) return 1.0;
    if (lower.includes('@')) return 0.5;
    return 0.4;
  }

  const clamp01 = (x: number) => Math.max(0, Math.min(1, x));
  const avg = (xs: number[]) => xs.length === 0 ? 0 : xs.reduce((s, x) => s + x, 0) / xs.length;

  export async function recomputeConviction(principleId: number): Promise<number> {
    const p = Principles.get(principleId);
    if (!p) return 0;
    if (p.conviction_pinned) return p.conviction;
    const claims = Claims.listForPrinciple(principleId).filter(c => !c.is_rejected);
    if (claims.length === 0) {
      if (p.conviction !== 0) Principles.setConviction(principleId, 0);
      return 0;
    }
    const countTerm = Math.min(1.0, Math.log10(1 + claims.length) / 1.3);
    const auth = avg(claims.map(c => speakerAuthority(c.speaker)));
    const mod  = avg(claims.map(c => MODALITY_WEIGHT[c.modality] ?? 0.5));
    const ageDays = (Date.now() - new Date(p.last_observed_at + 'Z').getTime()) / 86400000;
    const recency = Math.exp(-ageDays / 365);
    const raw = countTerm * 0.4 + auth * 0.2 + mod * 0.2 + recency * 0.2;
    const next = clamp01(raw);
    const prior = p.conviction;
    if (Math.abs(next - prior) > 0.01) {
      Principles.setConviction(principleId, next);
      if (Math.abs(next - prior) > 0.1) {
        DriftEvents.create({
          principle_id: principleId,
          prior_conviction: prior, new_conviction: next, delta: next - prior,
          cause: 'new_claim',
        });
        recordEvent({
          actorKind: 'method', actorId: 'conviction_v1',
          action: ACTIONS.PRINCIPLE_UPDATE,
          subjectKind: 'principle', subjectId: String(principleId),
          payload: { priorConviction: prior, newConviction: next, claims: claims.length },
        });
      }
    }
    return next;
  }

  export function pinConviction(principleId: number, value: number, byUserId: string): void {
    const v = clamp01(value);
    Principles.pinConviction(principleId, v, byUserId);
    recordEvent({
      actorKind: 'user', actorId: byUserId,
      action: ACTIONS.PRINCIPLE_PIN,
      subjectKind: 'principle', subjectId: String(principleId),
      payload: { value: v },
    });
  }

  export function unpinConviction(principleId: number, byUserId: string): void {
    Principles.unpinConviction(principleId, byUserId);
    recordEvent({
      actorKind: 'user', actorId: byUserId,
      action: ACTIONS.PRINCIPLE_UPDATE,
      subjectKind: 'principle', subjectId: String(principleId),
      payload: { unpin: true },
    });
  }

2. Calling sites: recomputeConviction is called from
   principle_cluster_v1 (already wired in prompt 06). It is ALSO
   called whenever a Claim is rejected (Claims.markRejected wrapper)
   and after a principle merge.

ACCEPTANCE CRITERIA
- A Principle backed by 1 Colin assertion (today) → conviction in
  [0.5, 0.7].
- The same principle after 20 Colin assertions → > 0.85.
- A 3-year-old principle with no recent reinforcement → recency term
  < 0.05, total conviction < 0.4.
- Pinning a Principle to 0.9 makes downstream consumers see 0.9 even
  though the formula would say otherwise.
