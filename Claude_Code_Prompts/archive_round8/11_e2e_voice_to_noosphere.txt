### 2. `tests/e2e/test_dialectic_to_noosphere.py`

```python
@pytest.mark.e2e
def test_recording_lands_as_conclusions_in_codex(
    seed_audio_fixture, fake_codex_and_supabase, sqlite_codex_with_api_hookup, monkeypatch,
):
    """Full happy path:
       record.wav (fixture) → trim → transcribe → title → upload → codex → ingest-from-codex → Conclusions
    """
    # ── 1. Set Dialectic to use the mock server ──
    monkeypatch.setenv("DIALECTIC_CLOUD_URL", fake_codex_and_supabase.base_url)
    monkeypatch.setenv("DIALECTIC_CLOUD_API_KEY", "test-key")

    # ── 2. Build a RecordingArtifact pointing at the seed wav ──
    from dialectic.recording_pipeline import RecordingArtifact, RecordingPipeline
    artifact = RecordingArtifact(
        audio_path=seed_audio_fixture,
        duration_seconds=30.0, sample_rate=16000, channels=1,
    )
    # ── 3. Run the pipeline synchronously (no Qt loop) ──
    pipeline = RecordingPipeline(artifact)
    # Gate real whisper to avoid 2GB download in CI; mock the transcribe stage to return
    # a canonical transcript that exercises downstream stages.
    monkeypatch.setattr(
        "dialectic.recording_pipeline.RecordingPipeline._stage_transcribe",
        lambda self: self._fake_transcribe(
            "We were discussing the purpose of the school. The claim is that "
            "inquiry matters more than credentialing. A second claim: narrative "
            "priced assets overpay by about eighteen percent."
        ),
    )
    # Implement `_fake_transcribe` as a helper on the pipeline (or inline the monkeypatch).
    pipeline.run()
    assert pipeline.result.upload_id is not None, "upload failed"
    upload_id = pipeline.result.upload_id

    # ── 4. Run ingest-from-codex against the SQLite fake ──
    from noosphere.codex_bridge import ingest_from_codex
    result = ingest_from_codex(
        upload_id=upload_id, use_llm=False, dry_run=False,
        codex_db_url=sqlite_codex_with_api_hookup.url,
    )
    assert result.num_conclusions_written >= 2

    # ── 5. Assert the Conclusion text contains at least one phrase from the transcript ──
    rows = sqlite_codex_with_api_hookup.conn.execute(
        'SELECT "conclusionText" FROM "Conclusion" WHERE "uploadId"=?',
        (upload_id,),
    ).fetchall()
    texts = [r[0].lower() for r in rows]
    assert any("inquiry" in t or "credential" in t or "narrative" in t for t in texts), \
        "expected at least one conclusion to reference the transcript's substantive content"

    # ── 6. Assert extractionMethod on the Upload row ──
    r = sqlite_codex_with_api_hookup.conn.execute(
        'SELECT "extractionMethod", status FROM "Upload" WHERE id=?', (upload_id,),
    ).fetchone()
    assert r[0].startswith("dialectic-")
    assert r[1] == "ingested"
```

```python
@pytest.mark.e2e
def test_original_m4a_bug_does_not_regress(
    fake_codex_and_supabase, sqlite_codex_with_api_hookup, tiny_audio_fixture, monkeypatch,
):
    """Reproduces the exact April 2026 failure: m4a upload, textContent=NULL,
    no transcript pre-attached. Must complete through the extractor."""
    # Insert the upload directly, bypassing Dialectic
    upload_id = sqlite_codex_with_api_hookup.insert_upload(
        mime="audio/mp4", text=None, file_path=str(tiny_audio_fixture),
        title="The Purpose of The School: Theseus Discussion April 18, 2026",
    )
    # Mock the AudioExtractor to avoid a real whisper call
    monkeypatch.setattr(
        "noosphere.extractors.audio_extractor.AudioExtractor.extract",
        lambda self, c: ExtractedText(
            text="the purpose of the school is inquiry not credentialing",
            source_format="audio/mp4",
            extraction_method="faster-whisper",
        ),
    )
    from noosphere.codex_bridge import ingest_from_codex
    result = ingest_from_codex(
        upload_id=upload_id, use_llm=False, dry_run=False,
        codex_db_url=sqlite_codex_with_api_hookup.url,
    )
    # It did NOT raise "has no textContent"
    assert result.num_conclusions_written >= 1
    r = sqlite_codex_with_api_hookup.conn.execute(
        'SELECT status, "extractionMethod", "errorMessage" FROM "Upload" WHERE id=?',
        (upload_id,),
    ).fetchone()
    assert r[0] == "ingested"
    assert r[1] == "faster-whisper"
    assert r[2] is None  # no error
```

### 4. `dialectic/tests/test_pipeline_invariants.py`

Five invariants, one test each:

1. **Pipeline stages run in order `trim, transcribe, title, upload`.** Capture `stage_started` emissions; assert equals `["trim", "transcribe", "title", "upload"]`.
2. **Upload is always on the trimmed file, never the raw.** With a real `auto_trim` pass (or mocked to return a specific `trimmed_audio` path), assert `codex_upload.upload_recording` receives that exact path.
3. **Title is never empty or `None`.** Even in the full-fallback path (LLM mocked to raise), `pipeline.result.title` is a non-empty string starting with `"Dialectic session"`.
4. **Offline upload queues.** With the Codex mock set to 500 on prepare, run the pipeline; assert `pending_uploads.jsonl` gets exactly one line matching the recording; the modal reports `stage_failed("upload", ...)` with the retry hint string in it.
5. **Transcript and `textContent` stay aligned.** When `_stage_transcribe` produces text X, the final row in the Codex has `textContent == X`. Divergence between these is a category of silent failure we must not allow.
