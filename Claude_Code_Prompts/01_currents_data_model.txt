### 1. Add enums and Pydantic models

Append to `noosphere/noosphere/models.py` (keep the existing import/ordering conventions — these enums are `str` subclasses, these models inherit from `StrictModel` where relevant):

```python
class CurrentEventSource(str, Enum):
    """Where the raw event came from. X is the only live source in round 7;
    keep the enum open so RSS / GDELT / academic feeds can be added later."""
    X_POST = "x_post"
    X_THREAD = "x_thread"
    MANUAL = "manual"  # operator-seeded for testing or editorial override


class CurrentEventStatus(str, Enum):
    OBSERVED = "observed"        # ingested, awaiting relevance gate
    ABSTAINED = "abstained"      # relevance gate said "not enough Noosphere coverage"
    OPINED = "opined"            # EventOpinion generated
    SUPPRESSED = "suppressed"    # operator-hidden (safety / abuse / legal)


class OpinionStance(str, Enum):
    """Coarse stance label for UI filtering — NOT the opinion text."""
    AGREES = "agrees"            # event aligns with firm's prior conclusions
    DISAGREES = "disagrees"      # event conflicts with firm's prior conclusions
    COMPLICATES = "complicates"  # event refines, qualifies, or reframes
    INSUFFICIENT = "insufficient"  # sources don't support a stance (should be rare; abstain path preferred)


class CurrentEvent(StrictModel):
    id: str
    source: CurrentEventSource
    source_url: str                 # canonical URL to the X post or thread root
    source_author_handle: str       # e.g., "@someone"
    source_captured_at: datetime    # when it was posted (per X)
    ingested_at: datetime           # when we fetched it
    raw_text: str                   # the post / thread body, concatenated if thread
    dedupe_hash: str                # stable hash over normalized source_url + content
    embedding: Optional[list[float]] = None   # populated in prompt 03
    topic_hint: Optional[str] = None          # populated in prompt 03
    status: CurrentEventStatus = CurrentEventStatus.OBSERVED
    status_reason: Optional[str] = None       # e.g., "no sources above threshold"


class EventOpinion(StrictModel):
    id: str
    event_id: str                   # FK → CurrentEvent.id
    generator_model: str            # e.g., "claude-haiku-4-5"
    generated_at: datetime
    stance: OpinionStance
    confidence: float               # 0..1, self-reported by the LLM then calibration-adjusted later
    headline: str                   # 1-sentence summary shown on the card
    body_markdown: str              # full opinion text, markdown, typically 3–8 sentences
    uncertainty_notes: list[str]    # explicit flags: "sources are dated", "only 2 citations", etc.
    sources_considered: int         # how many retrieval hits fed the generation
    sources_cited: int              # how many made it into OpinionCitation rows (must match row count)
    generator_tokens_prompt: int
    generator_tokens_completion: int
    revoked: bool = False           # operator kill-switch
    revoked_reason: Optional[str] = None


class OpinionCitation(StrictModel):
    """A single source anchor inside an EventOpinion.

    Exactly one of (conclusion_id, claim_id) is non-null — the opinion
    is cited against either a published Conclusion or a raw founder/voice
    Claim. Principles are referenced via their underlying Claim id.
    """
    id: str
    opinion_id: str
    conclusion_id: Optional[str] = None
    claim_id: Optional[str] = None
    quoted_span: str                # verbatim text from the source shown in the UI
    relevance_score: float          # retrieval score carried forward, 0..1
    ordinal: int                    # ordering inside the opinion (0, 1, 2, …)


class FollowUpSession(StrictModel):
    id: str
    opinion_id: str                 # root EventOpinion
    created_at: datetime
    last_activity_at: datetime
    expires_at: datetime            # ingester writes now + 24h
    client_fingerprint: str         # sha256 over (IP, UA, day) — for rate limiting, not identification
    message_count: int = 0


class FollowUpMessageRole(str, Enum):
    USER = "user"
    ASSISTANT = "assistant"
    SYSTEM = "system"


class FollowUpMessage(StrictModel):
    id: str
    session_id: str
    role: FollowUpMessageRole
    created_at: datetime
    content: str
    # Only meaningful when role == ASSISTANT:
    citations: list[OpinionCitation] = Field(default_factory=list)
    tokens_prompt: int = 0
    tokens_completion: int = 0
    refused: bool = False           # true if the assistant declined (e.g., no sources)
    refusal_reason: Optional[str] = None
```

- Use `datetime` from stdlib — match the existing file's imports.
- Ensure `Optional` and `Field` imports are already present; if not, add them to the existing import block.
- Every new model inherits from `StrictModel`. Every list field uses `Field(default_factory=list)`.
