4. **Create `founder-portal/desktop/updater.ts`** — auto-update module using `electron-updater`:
   ```typescript
   import { autoUpdater } from 'electron-updater';
   import { BrowserWindow, dialog } from 'electron';
   import log from 'electron-log';

   autoUpdater.logger = log;

   export function initAutoUpdater(mainWindow: BrowserWindow): void {
     // Check for updates on launch (silently)
     autoUpdater.checkForUpdatesAndNotify();

     autoUpdater.on('update-available', (info) => {
       dialog.showMessageBox(mainWindow, {
         type: 'info',
         title: 'Update Available',
         message: `Version ${info.version} is available. It will be downloaded in the background.`,
       });
     });

     autoUpdater.on('update-downloaded', (info) => {
       dialog.showMessageBox(mainWindow, {
         type: 'info',
         title: 'Update Ready',
         message: `Version ${info.version} has been downloaded. Restart to apply the update?`,
         buttons: ['Restart', 'Later'],
       }).then((result) => {
         if (result.response === 0) {
           autoUpdater.quitAndInstall();
         }
       });
     });

     autoUpdater.on('error', (err) => {
       log.error('Auto-updater error:', err);
     });
   }
   ```
   Add `electron-updater` and `electron-log` to devDependencies in `package.json`.

5. **Modify `founder-portal/desktop/main.ts`** — add auto-updater integration:
