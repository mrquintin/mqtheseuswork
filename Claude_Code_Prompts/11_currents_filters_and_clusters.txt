## Step-by-step

### 1. `filterMatch.ts` — the match predicate

```ts
import type { PublicOpinion, Stance } from "./currentsTypes";

export interface FilterState {
  topic: string | null;        // exact match on topic_hint
  stance: Stance | null;
  q: string | null;            // substring match, case-insensitive, over headline+body
  since: string | null;        // ISO date; opinions generated_at >= since
  view: "chronological" | "by-topic";
}

export const EMPTY_FILTER: FilterState = {
  topic: null, stance: null, q: null, since: null, view: "chronological",
};

export function matches(op: PublicOpinion, f: FilterState): boolean {
  if (f.topic && op.topic_hint !== f.topic) return false;
  if (f.stance && op.stance !== f.stance) return false;
  if (f.since && op.generated_at < f.since) return false;
  if (f.q) {
    const q = f.q.toLowerCase();
    const hay = (op.headline + " " + op.body_markdown).toLowerCase();
    if (!hay.includes(q)) return false;
  }
  return true;
}

export function filterToParams(f: FilterState): URLSearchParams {
  const p = new URLSearchParams();
  if (f.topic) p.set("topic", f.topic);
  if (f.stance) p.set("stance", f.stance);
  if (f.q) p.set("q", f.q);
  if (f.since) p.set("since", f.since);
  if (f.view !== "chronological") p.set("view", f.view);
  return p;
}

export function paramsToFilter(sp: URLSearchParams | Record<string, string | undefined>): FilterState {
  const get = (k: string) => (sp instanceof URLSearchParams ? sp.get(k) : (sp[k] ?? null));
  return {
    topic: get("topic") || null,
    stance: (get("stance") as Stance | null) || null,
    q: get("q") || null,
    since: get("since") || null,
    view: (get("view") as FilterState["view"]) || "chronological",
  };
}
```

### 2. `FilterBar.tsx`

Sticky header below the live banner. Four controls, minimal chrome:

```tsx
"use client";

import { useRouter, useSearchParams, usePathname } from "next/navigation";
import { useMemo } from "react";
import { filterToParams, paramsToFilter, type FilterState } from "@/lib/filterMatch";

const STANCES = ["agrees", "disagrees", "complicates", "insufficient"] as const;
const SINCE_PRESETS: { label: string; hours: number | null }[] = [
  { label: "any time", hours: null },
  { label: "1 h", hours: 1 },
  { label: "24 h", hours: 24 },
  { label: "7 d", hours: 24 * 7 },
];

export function FilterBar({ topics }: { topics: string[] }) {
  const router = useRouter();
  const pathname = usePathname();
  const sp = useSearchParams();
  const f = useMemo(() => paramsToFilter(sp), [sp]);

  const update = (next: Partial<FilterState>) => {
    const merged = { ...f, ...next };
    const qs = filterToParams(merged).toString();
    router.replace(qs ? `${pathname}?${qs}` : pathname, { scroll: false });
  };

  return (
    <div style={{
      position: "sticky", top: 0, zIndex: 5,
      background: "var(--currents-bg)",
      borderBottom: "1px solid var(--currents-border)",
      padding: "0.75rem 0", marginBottom: "1rem",
      display: "flex", flexWrap: "wrap", gap: "0.7rem", alignItems: "center",
    }}>
      {/* search */}
      <input
        type="search"
        placeholder="search opinions"
        defaultValue={f.q ?? ""}
        onChange={(e) => update({ q: e.target.value || null })}
        style={{
          background: "var(--currents-surface)",
          border: "1px solid var(--currents-border)",
          color: "var(--currents-parchment)",
          padding: "0.35rem 0.6rem",
          borderRadius: 2,
          fontSize: "0.85rem",
          minWidth: 180,
        }}
      />

      {/* topic dropdown */}
      <select
        value={f.topic ?? ""}
        onChange={(e) => update({ topic: e.target.value || null })}
        style={chipSelectStyle}
      >
        <option value="">all topics</option>
        {topics.map(t => <option key={t} value={t}>{t}</option>)}
      </select>

      {/* stance chips */}
      <div style={{ display: "flex", gap: "0.25rem" }}>
        {STANCES.map(s => (
          <button
            key={s}
            onClick={() => update({ stance: f.stance === s ? null : s })}
            style={{
              ...chipBase,
              background: f.stance === s ? "var(--currents-gold)" : "var(--currents-surface)",
              color: f.stance === s ? "#000" : "var(--currents-parchment-dim)",
            }}
          >
            {s}
          </button>
        ))}
      </div>

      {/* since presets */}
      <div style={{ display: "flex", gap: "0.25rem", marginLeft: "auto" }}>
        {SINCE_PRESETS.map(p => {
          const iso = p.hours == null ? null : new Date(Date.now() - p.hours * 3600_000).toISOString();
          const active = (f.since ?? null) === iso;
          return (
            <button key={p.label} onClick={() => update({ since: iso })} style={{
              ...chipBase,
              background: active ? "var(--currents-gold)" : "var(--currents-surface)",
              color: active ? "#000" : "var(--currents-parchment-dim)",
            }}>
              {p.label}
            </button>
          );
        })}
      </div>

      {/* view toggle */}
      <button
        onClick={() => update({ view: f.view === "chronological" ? "by-topic" : "chronological" })}
        style={chipBase}
      >
        view: {f.view === "chronological" ? "chronological" : "by topic"}
      </button>
    </div>
  );
}
```

### 4. `TopicClusters.tsx`

```tsx
"use client";

import { useMemo, useState } from "react";
import type { PublicOpinion } from "@/lib/currentsTypes";
import { OpinionCard } from "./OpinionCard";

export function TopicClusters({ opinions }: { opinions: PublicOpinion[] }) {
  const groups = useMemo(() => {
    const by = new Map<string, PublicOpinion[]>();
    for (const op of opinions) {
      const key = op.topic_hint || "unclassified";
      const arr = by.get(key) ?? [];
      arr.push(op);
      by.set(key, arr);
    }
    return [...by.entries()].sort((a, b) => b[1].length - a[1].length);
  }, [opinions]);

  return (
    <div>
      {groups.map(([topic, ops]) => (
        <TopicGroup key={topic} topic={topic} ops={ops} />
      ))}
    </div>
  );
}

function TopicGroup({ topic, ops }: { topic: string; ops: PublicOpinion[] }) {
  const [open, setOpen] = useState(true);
  return (
    <section style={{ marginBottom: "1.4rem" }}>
      <button onClick={() => setOpen(o => !o)} style={{
        width: "100%", textAlign: "left",
        background: "transparent",
        border: "none",
        padding: "0.4rem 0",
        fontSize: "0.9rem",
        color: "var(--currents-gold)",
        letterSpacing: "0.05em",
        textTransform: "uppercase",
        cursor: "pointer",
      }}>
        {open ? "▾" : "▸"} {topic} <span style={{ color: "var(--currents-muted)" }}>({ops.length})</span>
      </button>
      {open && ops.map(op => <OpinionCard key={op.id} op={op} />)}
    </section>
  );
}
```

### 5. Server `page.tsx` reads search params

```tsx
import { listCurrents } from "@/lib/currentsApi";
import { FeedClient } from "./FeedClient";
import { paramsToFilter } from "@/lib/filterMatch";

export const dynamic = "force-dynamic";

export default async function CurrentsPage({
  searchParams,
}: { searchParams: Promise<Record<string, string | undefined>> }) {
  const sp = await searchParams;
  const filter = paramsToFilter(sp);
  let seed = [];
  try {
    const resp = await listCurrents({
      limit: 20,
      topic: filter.topic ?? undefined,
      stance: filter.stance ?? undefined,
      since: filter.since ?? undefined,
    });
    seed = resp.items;
  } catch (err) {
    console.error("currents_initial_fetch_failed", err);
  }
  return <FeedClient seed={seed} initialFilter={filter} />;
}
```
