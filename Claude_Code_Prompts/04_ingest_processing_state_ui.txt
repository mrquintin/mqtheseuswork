### 4. `UploadRowDetail.tsx` — failure detail + retry

```tsx
export function UploadRowDetail({ upload }: { upload: UploadListItem }) {
  if (upload.status === "ingested" && upload.extractionMethod) {
    return (
      <div className="muted" style={{ fontSize: "0.78rem" }}>
        {humanMethod(upload.extractionMethod)}
      </div>
    );
  }
  if (upload.status !== "failed") return null;
  return (
    <details style={{ fontSize: "0.82rem", marginTop: "0.25rem" }}>
      <summary style={{ cursor: "pointer", color: "var(--error)" }}>
        {truncate(upload.errorMessage ?? "unknown error", 140)}
      </summary>
      <pre style={{ whiteSpace: "pre-wrap", marginTop: "0.35rem",
                    padding: "0.5rem", background: "#fdf7f6",
                    border: "1px solid #e5b6ae", borderRadius: 2 }}>
        {upload.errorMessage}
      </pre>
      <RetryButton uploadId={upload.id} />
    </details>
  );
}

function humanMethod(m: string): string {
  const map: Record<string, string> = {
    "passthrough": "stored as text",
    "faster-whisper": "transcribed locally (faster-whisper)",
    "openai-whisper-1": "transcribed via OpenAI",
    "pypdf": "extracted with pypdf",
    "ocrmypdf": "OCR'd with ocrmypdf",
  };
  return map[m] ?? `extracted via ${m}`;
}
```

### 5. `RetryButton`

A small client component that POSTs to `/api/upload/[id]/retry`. On success, optimistically updates the visible status to `pending` and triggers a list refetch. On failure, surfaces the error inline.

### 6. `api/upload/[id]/retry/route.ts`

```ts
export async function POST(_req: NextRequest, { params }: { params: { id: string } }) {
  const session = await getServerSession(authOptions);
  if (!session?.user?.organizationId) return NextResponse.json({ error: "auth" }, { status: 401 });

  const upload = await prisma.upload.findUnique({ where: { id: params.id } });
  if (!upload) return NextResponse.json({ error: "not_found" }, { status: 404 });
  if (upload.organizationId !== session.user.organizationId) {
    return NextResponse.json({ error: "forbidden" }, { status: 403 });
  }
  if (upload.status !== "failed") {
    return NextResponse.json(
      { error: "only failed uploads can be retried", currentStatus: upload.status },
      { status: 409 },
    );
  }

  await prisma.upload.update({
    where: { id: params.id },
    data: { status: "pending", errorMessage: null, extractionMethod: null },
  });
  return NextResponse.json({ ok: true });
}
```

### 7. Noosphere-side: write `extractionMethod` on success

In `codex_bridge.py`, after a successful extraction:

```python
cur.execute(
    'UPDATE "Upload" SET "extractionMethod" = %s WHERE id = %s',
    (extracted.extraction_method, upload_id),

```

and on the intermediate states:

- After fetch, before extraction: `SET status = 'extracting'`
- After extraction, before claim extraction: `SET status = 'awaiting_ingest'`
- Entering claim extraction: `SET status = 'processing'`  (this matches today's code)
- Success: `SET status = 'ingested', "extractionMethod" = ...`

## Prohibitions

- Do NOT poll the dashboard's status column more often than once per 8 seconds. The noosphere runner works in multi-second cycles; faster polling is noise.
- Do NOT show the raw stack trace of extraction failures to the founder. `errorMessage` as set by codex_bridge is already human-scoped. Raw tracebacks go to the noosphere runner's logs.
- Do NOT allow retry on statuses other than `failed`. Retrying `processing` could race with the runner and double-write Conclusions.
- Do NOT rename any existing status. `pending`, `processing`, `ingested`, `failed` are load-bearing strings in every prior migration.
