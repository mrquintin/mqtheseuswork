### 2. `useLiveOpinions` hook

```typescript
// theseus-codex/src/lib/useLiveOpinions.ts
"use client";

import { useEffect, useReducer, useRef } from "react";
import type { PublicOpinion } from "./currentsTypes";

interface State {
  opinions: PublicOpinion[];
  connected: boolean;
}

export function useLiveOpinions(seed: PublicOpinion[]) {
  const [state, dispatch] = useReducer(reducer, { opinions: seed, connected: false });
  const reconnectMs = useRef(1000);
  useEffect(() => {
    let cancelled = false; let es: EventSource | null = null;
    const connect = () => {
      es = new EventSource("/api/currents/stream");
      es.onopen = () => { dispatch({ type: "connected" }); reconnectMs.current = 1000; };
      es.onmessage = (e) => {
        try { dispatch({ type: "opinion", payload: JSON.parse(e.data) }); } catch {}
      };
      es.onerror = () => {
        es?.close();
        dispatch({ type: "disconnected" });
        if (cancelled) return;
        const wait = Math.min(reconnectMs.current, 30_000);
        reconnectMs.current = Math.min(reconnectMs.current * 2, 30_000);
        setTimeout(connect, wait);
      };
    };
    connect();
    return () => { cancelled = true; es?.close(); };
  }, []);
  return state;
}

// Reducer: prepends new opinions, dedupes by id, caps at 200 in-memory
```
