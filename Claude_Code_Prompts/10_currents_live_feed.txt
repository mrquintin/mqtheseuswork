### 2. `relativeTime.ts`

```ts
export function relativeTime(iso: string, now = Date.now()): string {
  const t = new Date(iso).getTime();
  const s = Math.max(1, Math.round((now - t) / 1000));
  if (s < 60) return `${s}s ago`;
  const m = Math.round(s / 60); if (m < 60) return `${m}m ago`;
  const h = Math.round(m / 60); if (h < 24) return `${h}h ago`;
  const d = Math.round(h / 24); return `${d}d ago`;
}
```

### 3. `OpinionCard.tsx`

```tsx
"use client";

import Link from "next/link";
import type { PublicOpinion } from "@/lib/currentsTypes";
import { STANCE_COLOR, STANCE_LABEL, confidenceBand } from "@/lib/stanceStyles";
import { renderSafeMarkdown } from "@/lib/safeMarkdown";
import { relativeTime } from "@/lib/relativeTime";

export function OpinionCard({ op }: { op: PublicOpinion }) {
  const stanceColor = STANCE_COLOR[op.stance];
  const band = confidenceBand(op.confidence);
  const visibleCites = op.citations.slice(0, 3);
  const moreCites = op.citations.length - visibleCites.length;

  return (
    <article
      className="currents-fade-in"
      style={{
        background: "var(--currents-bg-elevated)",
        border: "1px solid var(--currents-border)",
        borderLeft: `3px solid ${stanceColor}`,
        borderRadius: 3,
        padding: "1.1rem 1.25rem",
        marginBottom: "0.9rem",
      }}
    >
      {/* metadata row */}
      <div style={{
        display: "flex", justifyContent: "space-between", alignItems: "center",
        gap: "0.75rem", marginBottom: "0.55rem",
        fontSize: "0.72rem",
        color: "var(--currents-parchment-dim)",
        letterSpacing: "0.04em",
        textTransform: "uppercase",
      }}>
        <div style={{ display: "flex", gap: "0.6rem", alignItems: "center" }}>
          <span style={{
            background: stanceColor,
            color: "#000",
            padding: "0.12rem 0.5rem",
            borderRadius: 2,
            fontWeight: 600,
            letterSpacing: "0.02em",
          }}>
            {STANCE_LABEL[op.stance]}
          </span>
          <span title={`confidence ${op.confidence.toFixed(2)}`}>
            conf · {band}
          </span>
          {op.topic_hint && <span>topic · {op.topic_hint}</span>}
        </div>
        <span>{relativeTime(op.generated_at)}</span>
      </div>

      {/* headline */}
      <h2 style={{
        margin: 0,
        fontFamily: "'EB Garamond', serif",
        fontSize: "1.15rem",
        lineHeight: 1.3,
        color: "var(--currents-parchment)",
      }}>
        <Link href={`/currents/${op.id}`} style={{ color: "inherit" }}>
          {op.headline}
        </Link>
      </h2>

      {/* body */}
      <div style={{
        margin: "0.55rem 0 0.7rem",
        fontSize: "0.93rem",
        lineHeight: 1.55,
        color: "var(--currents-parchment)",
      }}>
        {renderSafeMarkdown(op.body_markdown)}
      </div>

      {/* uncertainty notes */}
      {op.uncertainty_notes.length > 0 && (
        <div style={{
          fontSize: "0.75rem",
          color: "var(--currents-amber)",
          fontStyle: "italic",
          marginBottom: "0.5rem",
        }}>
          {op.uncertainty_notes.slice(0, 3).map((n, i) => (
            <span key={i}>{n}{i < op.uncertainty_notes.length - 1 ? " · " : ""}</span>
          ))}
        </div>
      )}

      {/* source strip + actions */}
      <div style={{
        display: "flex", justifyContent: "space-between", alignItems: "center",
        gap: "0.6rem", flexWrap: "wrap",
        paddingTop: "0.6rem",
        borderTop: "1px solid var(--currents-border)",
      }}>
        <div style={{ display: "flex", gap: "0.35rem", flexWrap: "wrap" }}>
          {visibleCites.map((c) => (
            <Link
              key={c.source_id}
              href={`/currents/${op.id}#src-${encodeURIComponent(c.source_id)}`}
              style={{
                fontSize: "0.72rem",
                padding: "0.18rem 0.55rem",
                border: "1px solid var(--currents-border)",
                borderRadius: 2,
                color: "var(--currents-parchment-dim)",
                textDecoration: "none",
              }}
              title={c.quoted_span}
            >
              {c.source_kind === "conclusion" ? "⸺ conclusion" : "⸺ claim"}
            </Link>
          ))}
          {moreCites > 0 && (
            <span style={{ fontSize: "0.72rem", color: "var(--currents-muted)" }}>
              +{moreCites} more
            </span>
          )}
        </div>
        <div style={{ display: "flex", gap: "0.8rem", fontSize: "0.78rem" }}>
          <a href={op.event_source_url} target="_blank" rel="noopener noreferrer"
             style={{ color: "var(--currents-parchment-dim)" }}>
            {op.event_author_handle} ↗
          </a>
          <Link href={`/currents/${op.id}#ask`} style={{ color: "var(--currents-gold)" }}>
            Ask a follow-up →
          </Link>
        </div>
      </div>
    </article>
  );
}
```

### 5. `FeedClient.tsx`

```tsx
"use client";

import { useLiveOpinions } from "@/lib/useLiveOpinions";
import type { PublicOpinion } from "@/lib/currentsTypes";
import { OpinionCard } from "./OpinionCard";
import { LiveBanner } from "./LiveBanner";

export function FeedClient({ seed }: { seed: PublicOpinion[] }) {
  const { opinions, connected } = useLiveOpinions(seed);
  const lastOpinionAt = opinions[0]?.generated_at ?? null;

  return (
    <>
      <LiveBanner connected={connected} lastOpinionAt={lastOpinionAt} />
      {opinions.length === 0 ? (
        <EmptyState />
      ) : (
        opinions.map(op => <OpinionCard key={op.id} op={op} />)
      )}
    </>
  );
}

function EmptyState() {
  return (
    <div style={{
      border: "1px dashed var(--currents-border)",
      padding: "2rem",
      borderRadius: 3,
      color: "var(--currents-parchment-dim)",
      fontStyle: "italic",
      textAlign: "center",
    }}>
      No opinions yet. The firm abstains when its knowledge base does not support a position.
      Keep this page open — new items will arrive as events break.
    </div>
  );
}
```

### 6. `page.tsx` — server component that seeds initial data

```tsx
import { listCurrents } from "@/lib/currentsApi";
import { FeedClient } from "./FeedClient";

export const dynamic = "force-dynamic";
export const revalidate = 0;

export default async function CurrentsPage() {
  let seed = [];
  try {
    const resp = await listCurrents({ limit: 20 });
    seed = resp.items;
  } catch (err) {
    // Surface the failure clearly rather than rendering an empty page.
    console.error("currents_initial_fetch_failed", err);
  }
  return <FeedClient seed={seed} />;
}
```
