### SSE pass-through

```ts
export const runtime = "nodejs";
export const dynamic = "force-dynamic";

export async function GET(req: NextRequest, { params }: { params: { id: string } }) {
  const upstream = await fetch(`${BACKEND}/v1/currents/${params.id}/follow-up`, {
    method: "POST",
    headers: { ...passThroughHeaders(req), "x-client-id": fingerprintFor(req) },
    body: req.body,
    duplex: "half",
  });
  return new Response(upstream.body, {
    headers: {
      "content-type": "text/event-stream",
      "cache-control": "no-cache, no-transform",
      "connection": "keep-alive",
    },
  });
}
```

`passThroughHeaders`: forward `user-agent`, `x-forwarded-for` (or first IP from `forwarded`), and not much else. NEVER forward cookies or auth headers.

`fingerprintFor`: `sha256(ip + ua + UTC-day).slice(0, 32)`. Used by the FastAPI rate-limiter as the "anonymous identity" for follow-up rate-limiting.
