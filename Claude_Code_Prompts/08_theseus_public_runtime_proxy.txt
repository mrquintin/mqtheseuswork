### 4. Route handlers — proxies

All handlers live in `theseus-public/src/app/api/currents/...`.

```ts
// route.ts (list)
export const dynamic = "force-dynamic";
const BACKEND = process.env.CURRENTS_API_URL!; // e.g., http://current_events_api:8088

export async function GET(req: Request) {
  const url = new URL(req.url);
  const backend = new URL("/v1/currents" + url.search, BACKEND);
  const resp = await fetch(backend, { headers: forwardHeaders(req) });
  return new Response(await resp.text(), { status: resp.status, headers: jsonHeaders() });
}
```

`forwardHeaders` copies only `x-forwarded-for`, `user-agent`, `x-client-id` (no cookies — this is an anonymous surface).

SSE handlers use a pass-through ReadableStream:

```ts
// stream/route.ts
export const dynamic = "force-dynamic";
export const runtime = "nodejs";

export async function GET(req: Request) {
  const backend = new URL("/v1/currents/stream", BACKEND);
  const upstream = await fetch(backend, { headers: { Accept: "text/event-stream", ...forwardHeaders(req) } });
  if (!upstream.body) return new Response("upstream closed", { status: 502 });
  return new Response(upstream.body, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      "Connection": "keep-alive",
      "X-Accel-Buffering": "no",
    },
  });
}
```

Same pattern for POST follow-up; forward the JSON body and pipe the SSE response back.
