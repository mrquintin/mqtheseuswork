## Your task

Add a post-ingestion pipeline stage that: (a) embeds each `OBSERVED` `CurrentEvent`, (b) marks it `SUPPRESSED` if it is a near-duplicate of another recent event (same story, different post), (c) assigns a `topic_hint` by embedding-nearest-neighbor against existing `Topic` centroids and/or existing `Conclusion` topics, and (d) runs a relevance gate that marks events as `ABSTAINED` with `status_reason="no sources above threshold"` when retrieval finds nothing substantive to opine on.

Events that survive all three gates remain `OBSERVED` and are picked up by the opinion generator in prompt 05. An abstention is not a failure — it is the system honoring design invariant #1: no opinion without sources.

### 1. `enrich.py` — embed, dedupe-by-similarity, topic-hint

```python
from __future__ import annotations

from dataclasses import dataclass
from datetime import datetime, timedelta, timezone
from typing import Optional

import numpy as np

from noosphere.embeddings import embed_text            # reuse the existing helper
from noosphere.models import CurrentEvent, CurrentEventStatus, Topic
from noosphere.observability import get_logger
from noosphere.store import Store

logger = get_logger(__name__)


# Tune empirically. Start here:
NEAR_DUPLICATE_COSINE = 0.92
TOPIC_ASSIGNMENT_COSINE = 0.35   # loose — we just need a rough topic label


def _cosine(a: np.ndarray, b: np.ndarray) -> float:
    denom = (np.linalg.norm(a) * np.linalg.norm(b)) or 1e-9
    return float(np.dot(a, b) / denom)


@dataclass
class EnrichmentResult:
    event_id: str
    was_duplicate_of: Optional[str]     # ID of the event we collapsed into
    topic_hint: Optional[str]           # assigned topic (or None)


def enrich_event(store: Store, event: CurrentEvent, *, now: Optional[datetime] = None) -> EnrichmentResult:
    """Embed, check for near-duplicates in the last 24h, assign topic hint, persist."""
    now = now or datetime.now(timezone.utc)
    if event.embedding is not None:
        logger.info("enrich_skip_already_enriched", event_id=event.id)
        return EnrichmentResult(event.id, None, event.topic_hint)

    vec = np.asarray(embed_text(event.raw_text), dtype=np.float32)

    # 1. Near-duplicate check against recent events
    lookback = now - timedelta(hours=24)
    recent_ids = store.list_current_event_ids(since=lookback, limit=500)
    for other_id in recent_ids:
        if other_id == event.id:
            continue
        other = store.get_current_event(other_id)
        if other is None or other.embedding is None:
            continue
        sim = _cosine(vec, np.asarray(other.embedding, dtype=np.float32))
        if sim >= NEAR_DUPLICATE_COSINE:
            store.update_current_event_status(
                event.id,
                CurrentEventStatus.SUPPRESSED,
                reason=f"near_duplicate_of:{other_id}",
            )
            logger.info("enrich_near_duplicate", event_id=event.id, of=other_id, sim=sim)
            return EnrichmentResult(event.id, other_id, other.topic_hint)

    # 2. Topic assignment — nearest Topic centroid (if Topics have embeddings)
    topic_hint = _assign_topic_hint(store, vec)

    # 3. Persist embedding + topic hint
    store.set_current_event_topic_and_embedding(event.id, topic_hint=topic_hint or "", embedding=vec.tolist())
    logger.info("enrich_ok", event_id=event.id, topic=topic_hint)
    return EnrichmentResult(event.id, None, topic_hint)


def _assign_topic_hint(store: Store, event_vec: np.ndarray) -> Optional[str]:
    """Return the nearest Topic's label if similarity exceeds threshold, else None.

    If Topics don't carry precomputed centroid embeddings in your store,
    fall back to embedding the topic label on the fly and caching for the run.
    """
    topics = _load_topics_with_embeddings(store)
    if not topics:
        return None
    best_label = None
    best_sim = -1.0
    for label, tvec in topics:
        sim = _cosine(event_vec, tvec)
        if sim > best_sim:
            best_sim = sim
            best_label = label
    return best_label if best_sim >= TOPIC_ASSIGNMENT_COSINE else None
```

### 2. `relevance.py` — the abstention gate

```python
from __future__ import annotations

from dataclasses import dataclass
from typing import Optional

from noosphere.currents.retrieval_adapter import retrieve_for_event  # defined in prompt 04
from noosphere.models import CurrentEvent, CurrentEventStatus
from noosphere.observability import get_logger
from noosphere.store import Store

logger = get_logger(__name__)

# Tune empirically.
MIN_SOURCES_FOR_OPINION = 2
MIN_TOP_SCORE = 0.55


@dataclass
class RelevanceResult:
    event_id: str
    passed: bool
    reason: Optional[str]
    sources_found: int
    top_score: float


def check_relevance(store: Store, event: CurrentEvent) -> RelevanceResult:
    if event.status != CurrentEventStatus.OBSERVED:
        return RelevanceResult(event.id, False, f"skipped_status:{event.status.value}", 0, 0.0)
    if event.embedding is None:
        return RelevanceResult(event.id, False, "skipped_no_embedding", 0, 0.0)

    hits = retrieve_for_event(store, event)   # stub interface — real impl in prompt 04
    if len(hits) < MIN_SOURCES_FOR_OPINION or (hits and hits[0].score < MIN_TOP_SCORE):
        store.update_current_event_status(
            event.id,
            CurrentEventStatus.ABSTAINED,
            reason="no_sources_above_threshold",
        )
        logger.info("relevance_abstain", event_id=event.id, hits=len(hits),
                    top_score=(hits[0].score if hits else 0.0))
        return RelevanceResult(event.id, False, "no_sources_above_threshold",
                               len(hits), hits[0].score if hits else 0.0)

    return RelevanceResult(event.id, True, None, len(hits), hits[0].score)
```
