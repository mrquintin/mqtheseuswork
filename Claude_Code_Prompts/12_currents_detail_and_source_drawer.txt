### 1. `highlight.ts` — safe, React-node based

```ts
import React from "react";

/** Returns a React fragment with the first case-insensitive, whitespace-normalized
 *  occurrence of `needle` wrapped in <mark>. Never uses innerHTML. */
export function highlightFirst(haystack: string, needle: string): React.ReactNode {
  if (!needle) return haystack;
  const normalize = (s: string) => s.toLowerCase().replace(/\s+/g, " ");
  const hayNorm = normalize(haystack);
  const needleNorm = normalize(needle);
  const idx = hayNorm.indexOf(needleNorm);
  if (idx < 0) return haystack;
  // Map normalized index back to the original string (approximate: advance char-by-char).
  let origStart = -1, consumed = 0;
  for (let i = 0; i < haystack.length; i++) {
    const ch = haystack[i];
    const normCh = normalize(ch);
    if (normCh.length === 0) continue;
    if (consumed === idx) { origStart = i; break; }
    consumed += normCh.length;
  }
  if (origStart < 0) return haystack;
  // Advance origEnd until we've consumed `needleNorm.length` normalized chars.
  let origEnd = origStart, consumed2 = 0;
  while (origEnd < haystack.length && consumed2 < needleNorm.length) {
    const ch = haystack[origEnd];
    const n = normalize(ch);
    if (n.length > 0) consumed2 += n.length;
    origEnd += 1;
  }
  return React.createElement(React.Fragment, null,
    haystack.slice(0, origStart),
    React.createElement("mark", {
      style: {
        background: "var(--currents-gold-glow)",
        color: "var(--currents-parchment)",
        padding: "0 0.15rem",
        borderRadius: 1,
      },
    }, haystack.slice(origStart, origEnd)),
    haystack.slice(origEnd),
  );
}
```

### 2. `SourceCard.tsx`

Shows one citation with its full source text, highlighted quote, and, for conclusions, a permalink.

```tsx
import Link from "next/link";
import type { PublicSource, PublicCitation } from "@/lib/currentsTypes";
import { highlightFirst } from "@/lib/highlight";

export function SourceCard({
  source, citation, scrollTargetRef, active,
}: {
  source: PublicSource;
  citation: PublicCitation;
  scrollTargetRef?: React.Ref<HTMLDivElement>;
  active: boolean;
}) {
  return (
    <div
      ref={scrollTargetRef}
      id={`src-${citation.source_id}`}
      style={{
        padding: "0.9rem 1rem",
        marginBottom: "0.75rem",
        border: `1px solid ${active ? "var(--currents-gold)" : "var(--currents-border)"}`,
        borderRadius: 3,
        background: "var(--currents-surface)",
        transition: "border-color 400ms",
      }}
    >
      <header style={{
        display: "flex", justifyContent: "space-between", gap: "0.5rem",
        fontSize: "0.7rem", color: "var(--currents-parchment-dim)",
        textTransform: "uppercase", letterSpacing: "0.04em",
        marginBottom: "0.35rem",
      }}>
        <span>
          {source.source_kind}
          {source.origin && ` · ${source.origin}`}
          {source.topic_hint && ` · ${source.topic_hint}`}
        </span>
        {source.permalink && (
          <Link href={source.permalink} style={{ color: "var(--currents-gold)" }}>
            full conclusion →
          </Link>
        )}
      </header>
      <div style={{
        fontSize: "0.86rem",
        lineHeight: 1.55,
        color: "var(--currents-parchment)",
        fontFamily: "'EB Garamond', serif",
      }}>
        {highlightFirst(source.full_text, citation.quoted_span)}
      </div>
      <footer style={{
        marginTop: "0.4rem",
        fontSize: "0.7rem",
        color: "var(--currents-muted)",
      }}>
        id {source.source_id} · retrieval score {citation.relevance_score.toFixed(2)}
      </footer>
    </div>
  );
}
```

### 3. `SourceDrawer.tsx`

Renders the list of source cards. If URL hash matches a card, scroll it into view and flash `active` for 1.4s.

```tsx
"use client";

import { useEffect, useRef, useState } from "react";
import type { PublicCitation, PublicSource } from "@/lib/currentsTypes";
import { SourceCard } from "./SourceCard";

export function SourceDrawer({ citations, sources }: {
  citations: PublicCitation[];
  sources: PublicSource[];
}) {
  const [activeId, setActiveId] = useState<string | null>(null);
  const refs = useRef(new Map<string, HTMLDivElement>());

  useEffect(() => {
    const scrollToHash = () => {
      const h = window.location.hash;
      if (!h.startsWith("#src-")) return;
      const id = decodeURIComponent(h.slice(5));
      const el = refs.current.get(id);
      if (!el) return;
      el.scrollIntoView({ behavior: "smooth", block: "center" });
      setActiveId(id);
      setTimeout(() => setActiveId(null), 1400);
    };
    scrollToHash();
    window.addEventListener("hashchange", scrollToHash);
    return () => window.removeEventListener("hashchange", scrollToHash);
  }, []);

  const sourcesById = new Map(sources.map(s => [s.source_id, s]));
  return (
    <aside aria-label="Sources" style={{
      background: "var(--currents-bg-elevated)",
      border: "1px solid var(--currents-border)",
      borderRadius: 3,
      padding: "1rem",
      maxHeight: "calc(100vh - 4rem)",
      overflowY: "auto",
      position: "sticky",
      top: "1rem",
    }}>
      <h3 style={{
        margin: "0 0 0.7rem",
        fontSize: "0.78rem",
        color: "var(--currents-parchment-dim)",
        letterSpacing: "0.1em",
        textTransform: "uppercase",
      }}>
        Sources cited ({citations.length})
      </h3>
      {citations.map(c => {
        const src = sourcesById.get(c.source_id);
        if (!src) return (
          <div key={c.source_id} style={{
            padding: "0.6rem", border: "1px dashed var(--currents-border)",
            color: "var(--currents-amber)", fontSize: "0.78rem", marginBottom: "0.5rem",
          }}>
            Missing source (id {c.source_id}). It may have been revoked since this opinion was generated.
          </div>
        );
        return (
          <SourceCard
            key={c.source_id}
            citation={c}
            source={src}
            active={activeId === c.source_id}
            scrollTargetRef={(el) => { if (el) refs.current.set(c.source_id, el); }}
          />
        );
      })}
    </aside>
  );
}
```
