CorpusSlicer proxy (exact)
═══════════════════════════════════════════════════════════════════════════════
class CorpusSlicer:
    def __init__(self, store, cut: TemporalCut):
        self._store = store
        self._cut = cut

    def get_artifact(self, artifact_id: str):
        obj = self._store.get_artifact(artifact_id)
        if obj is None: return None
        if obj.created_at > self._cut.as_of: raise EmbargoViolation(f"{artifact_id} is after {self._cut.as_of}")
        return obj

    # Similarly for iter_claims, iter_chunks, iter_embeddings, etc.
    # Any method on Store not relevant to read paths is not proxied (calling it raises AttributeError).
