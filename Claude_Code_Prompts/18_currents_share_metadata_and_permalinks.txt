## Your task

### 1. `generateMetadata` for `/currents/[id]/page.tsx`

```tsx
import type { Metadata } from "next";
import { getCurrent } from "@/lib/currentsApi";

export async function generateMetadata({ params }): Promise<Metadata> {
  try {
    const op = await getCurrent(params.id);
    const description = op.body_markdown.slice(0, 220).replace(/\s+/g, " ").trim() + "…";
    return {
      title: op.headline,
      description,
      openGraph: {
        title: op.headline, description, type: "article",
        siteName: "Theseus Codex",
        url: `${SITE}/currents/${op.id}`,
      },
      twitter: { card: "summary_large_image", title: op.headline, description },
    };
  } catch {
    return { title: "Current event" };
  }
}
```

### 2. `<CopyLinkButton>` on the detail page

```tsx
"use client";
export function CopyLinkButton({ opinionId }: { opinionId: string }) {
  const [copied, setCopied] = useState(false);
  const onClick = async () => {
    const url = `${window.location.origin}/currents/${encodeURIComponent(opinionId)}`;
    try {
      await navigator.clipboard.writeText(url);
      setCopied(true);
      setTimeout(() => setCopied(false), 1400);
    } catch {
      window.prompt("Copy this permalink:", url);
    }
  };
  return (
    <button onClick={onClick} className="…">
      {copied ? "permalink copied" : "copy permalink"}
    </button>
  );
}
```

Mounted in the detail page's header strip (small, muted, parchment-bordered).
