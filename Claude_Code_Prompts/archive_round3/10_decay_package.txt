Scheduler loop (exact)
class Scheduler:
    def __init__(self, store, budget_per_run: int = 50, warmup_days: int = 14,
                 thrashing_threshold: int = 5):
        ...

    def run_once(self) -> list[RevalidationResult]:
        now = utc_now()
        candidates = self._enumerate_candidates(now)     # objects whose compute_freshness says re-validate
        candidates = self._filter_warmup(candidates, now)
        candidates.sort(key=lambda c: self._priority(c), reverse=True)
        results = []
        for obj in candidates[: self.budget_per_run]:
            if self._thrashing(obj, now):
                self._escalate_to_human(obj)
                continue
            results.append(self._revalidate(obj))
        return results

    def _revalidate(self, obj):
        method = self._canonical_revalidation_method_for(obj)
        output = method(obj)        # routed through REGISTRY so ledgered
        if self._materially_differs(output, obj.prior_output):
            return RevalidationResult(object_id=obj.id, outcome="disagreement", ...)
        return RevalidationResult(object_id=obj.id, outcome="confirmed", ...)
